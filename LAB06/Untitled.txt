


TASK1
-- a. List all student details
SELECT * FROM students;

-- b. Display all courses that offer more than 3 credits
SELECT * FROM courses WHERE credits > 3;

-- c. Show all enrollments with student name, course name, and grade
SELECT s.name AS student_name, c.course_name, e.grade
FROM enrollments e
JOIN students s ON e.student_id = s.student_id
JOIN courses c ON e.course_id = c.course_id;




TASK2


-- a. Show each course with the number of enrolled students
SELECT c.course_name, COUNT(e.student_id) AS num_students
FROM courses c
JOIN enrollments e ON c.course_id = e.course_id
GROUP BY c.course_id;

-- b. Display average GPA of students per course
SELECT c.course_name, AVG(s.gpa) AS avg_gpa
FROM courses c
JOIN enrollments e ON c.course_id = e.course_id
JOIN students s ON e.student_id = s.student_id
GROUP BY c.course_id;



TASK3
-- a. Find students who enrolled after '2024-02-10'
SELECT DISTINCT s.*
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
WHERE e.enroll_date > '2024-02-10';

-- b. Retrieve names of students who got the highest GPA
SELECT name
FROM students
WHERE gpa = (SELECT MAX(gpa) FROM students);

-- b. Delete enrollments where the grade is NULL
DELETE FROM enrollments WHERE grade IS NULL;




TASK5

-- a. Create view
CREATE VIEW student_course_view AS
SELECT s.name AS student_name, c.course_name, e.grade
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
JOIN courses c ON c.course_id = e.course_id;

-- b. Create stored procedure to enroll a student
DELIMITER //

CREATE PROCEDURE enroll_student (
  IN sid INT, IN cid INT, IN date DATE, IN grd DECIMAL(3,2)
)
BEGIN
  INSERT INTO enrollments (student_id, course_id, enroll_date, grade)
  VALUES (sid, cid, date, grd);
END //

DELIMITER ;
